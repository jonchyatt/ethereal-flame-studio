**Create `src/lib/jarvis/vault/vaultHealth.ts`:**

```typescript
import { execSync } from 'child_process';

export interface VaultStatus {
  status: 'locked' | 'unlocked' | 'unauthenticated';
  serverUrl: string;
  userEmail: string;
}

/**
 * Check current Bitwarden vault status using BW_SESSION from env.
 * Returns 'unauthenticated' if BW_SESSION is not set.
 */
export function checkVaultHealth(): VaultStatus {
  const session = process.env.BW_SESSION;
  if (!session) {
    return { status: 'unauthenticated', serverUrl: '', userEmail: '' };
  }
  try {
    const raw = execSync(`bw status --session "${session}"`, {
      encoding: 'utf8',
      timeout: 10_000,
    });
    return JSON.parse(raw) as VaultStatus;
  } catch {
    return { status: 'locked', serverUrl: '', userEmail: '' };
  }
}

/**
 * Ensure vault is unlocked. If locked, re-unlock using BW_MASTER_PASSWORD env.
 * Returns BW_SESSION string on success.
 * Throws if vault is unauthenticated (not logged in) or BW_MASTER_PASSWORD not set.
 */
export function ensureVaultUnlocked(): string {
  const status = checkVaultHealth();
  if (status.status === 'unlocked') {
    return process.env.BW_SESSION!;
  }
  if (status.status === 'unauthenticated') {
    throw new Error('Vault not logged in. Run: bw login');
  }
  // status === 'locked' -- attempt re-unlock
  const masterPw = process.env.BW_MASTER_PASSWORD;
  if (!masterPw) {
    throw new Error('Vault locked and BW_MASTER_PASSWORD not set. Run: bw unlock');
  }
  const newSession = execSync('bw unlock --passwordenv BW_MASTER_PASSWORD --raw', {
    encoding: 'utf8',
    timeout: 30_000,
    env: { ...process.env, BW_MASTER_PASSWORD: masterPw },
  }).trim();
  process.env.BW_SESSION = newSession;
  return newSession;
}
```

**Create `scripts/unlock-vault.sh`:**

Bash script that checks vault status and outputs BW_SESSION for PM2 env injection. Must handle three states: unlocked (re-export session), locked (prompt unlock), unauthenticated (tell user to login).

```bash
#!/bin/bash
# scripts/unlock-vault.sh
# Run at Jarvis startup to ensure Bitwarden vault is unlocked.
# Outputs: BW_SESSION=<token> on success (for eval or .env injection)

set -e

if ! command -v bw &> /dev/null; then
  echo "ERROR: Bitwarden CLI not installed. Run: npm install -g @bitwarden/cli" >&2
  exit 1
fi

STATUS=$(bw status 2>/dev/null | node -e "process.stdin.on('data',d=>{try{console.log(JSON.parse(d).status)}catch{console.log('unknown')}})" 2>/dev/null)

case $STATUS in
  "unlocked")
    echo "Vault already unlocked" >&2
    # Export current session key
    SESSION=$(bw unlock --raw 2>/dev/null)
    echo "BW_SESSION=$SESSION"
    ;;
  "locked")
    echo "Vault locked. Unlocking..." >&2
    SESSION=$(bw unlock --raw)
    if [ -z "$SESSION" ]; then
      echo "ERROR: Failed to unlock vault" >&2
      exit 1
    fi
    echo "BW_SESSION=$SESSION"
    ;;
  "unauthenticated")
    echo "ERROR: Not logged in to Bitwarden. Run: bw login" >&2
    exit 1
    ;;
  *)
    echo "ERROR: Unknown vault status: $STATUS" >&2
    exit 1
    ;;
esac
```

Make the script executable. Use `jq`-free JSON parsing (node -e inline) since jq may not be on Windows PATH -- node is always available in this project.

Define the form-filler sub-agent with a private Bitwarden MCP server. This sub-agent will be used by Phase 14 (browser automation) to fill login forms. The parent agent never gets access to `mcp__bitwarden__*` tools.

```typescript
import { ensureVaultUnlocked } from './vaultHealth';

export const FORM_FILLER_AGENT_NAME = 'form-filler';

/**
 * Build the form-filler sub-agent definition with a private Bitwarden MCP server.
 * The Bitwarden MCP server is scoped to this sub-agent only -- the parent agent
 * never loads bitwarden tools into its namespace.
 *
 * Call ensureVaultUnlocked() to get a valid BW_SESSION before building the config.
 * The session token is resolved at definition time (not deferred via ${} syntax)
 * to ensure it propagates through SDK -> sub-agent -> MCP server process chain.
 */
export function getFormFillerAgent() {
  const session = ensureVaultUnlocked();

  return {
    [FORM_FILLER_AGENT_NAME]: {
      description: 'Fills web forms using vault credentials. Use for logins and form submissions requiring stored passwords.',
      prompt: `You are a secure form-filling agent. You retrieve credentials from Bitwarden vault and inject them into web forms.

CRITICAL SECURITY RULES:
1. NEVER repeat, echo, log, or include any credential values (usernames, passwords, TOTP codes) in your response text
2. When reporting results, say "Credentials retrieved and injected" -- never show actual values
3. Take a screenshot BEFORE and AFTER every form submission
4. Stop and report if you encounter CAPTCHAs, 2FA prompts, or unexpected security challenges
5. Only submit forms when the parent agent has received explicit user approval
6. If a credential is not found in the vault, report "Credential not found for [site name]" -- do not guess or fabricate`,
      tools: ['mcp__bitwarden__*', 'mcp__playwright__*'],
      model: 'sonnet',
      mcpServers: {
        bitwarden: {
          type: 'stdio',
          command: 'cmd',
          args: ['/c', 'npx', '-y', '@bitwarden/mcp-server'],
          env: { BW_SESSION: session },
        },
      },
    },
  };
}
```

Key details:
- Uses `cmd /c npx` pattern (Windows PM2 compatibility, matching .mcp.json notion/playwright pattern)
- Resolves BW_SESSION at call time via `ensureVaultUnlocked()`, not via deferred `${BW_SESSION}` string
- `model: 'sonnet'` for cost efficiency (form-filling is mechanical, not creative)
- Tools restricted to `mcp__bitwarden__*` and `mcp__playwright__*` only

**Create `src/lib/jarvis/vault/canaryTest.ts`:**

```typescript
/**
 * Canary test: verifies that no credential values leaked into parent conversation history.
 * Run after a vault retrieval + injection cycle to confirm sub-agent isolation works.
 */
export interface CanaryResult {
  passed: boolean;
  details: string;
}

/**
 * Scan an array of conversation message strings for a known credential value.
 * The test credential should be a real value from a test vault entry.
 *
 * @param parentConversationHistory - Array of message text strings from parent agent
 * @param testCredentialValue - A known credential value that was retrieved by the sub-agent
 * @returns CanaryResult with passed=true if credential was NOT found (good), false if leaked (bad)
 */
export function runCredentialCanaryTest(
  parentConversationHistory: string[],
  testCredentialValue: string,
): CanaryResult {
  if (!testCredentialValue || testCredentialValue.length < 4) {
    return { passed: false, details: 'INVALID: testCredentialValue must be at least 4 characters' };
  }
  for (let i = 0; i < parentConversationHistory.length; i++) {
    if (parentConversationHistory[i].includes(testCredentialValue)) {
      return {
        passed: false,
        details: `FAIL: Credential value "${testCredentialValue.slice(0, 3)}..." found in parent conversation at message index ${i}`,
      };
    }
  }
  return { passed: true, details: 'No credential values found in parent conversation history' };
}
```
