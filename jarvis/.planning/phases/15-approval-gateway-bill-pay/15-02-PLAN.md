**Create `src/lib/jarvis/workflows/billPayWorkflow.ts`:**

This is the first CONSUMER of the generic approval gateway. It orchestrates the multi-stage bill payment pipeline.

```typescript
/**
 * Bill Payment Workflow -- Jarvis v5.0
 *
 * Multi-stage orchestrator for end-to-end bill payment:
 *   Stage 1: Look up bill in Notion (get service link URL)
 *   Stage 2: Navigate to portal, authenticate via vault, fill payment form (form-filler sub-agent)
 *   Stage 3: Take pre-submission screenshot and request human approval
 *   Stage 4: On approval, submit the payment (browser-worker sub-agent)
 *   Stage 5: Capture receipt screenshot and update Notion bill status
 *
 * This module is the FIRST CONSUMER of the generic approval gateway.
 * It uses requestApproval() from approvalGateway.ts for the human approval step.
 *
 * Sub-agents are spawned via the Claude Agent SDK query() function with the agents
 * parameter from buildSubAgents(). The workflow decomposes into separate sub-agent
 * calls because sub-agents cannot be paused mid-execution.
 */

import { query } from '@anthropic-ai/claude-code';
import { buildSubAgents } from '../agents/subAgentRegistry';
import { requestApproval } from '../approval/approvalGateway';
import { sendTelegramAlert } from '../telegram/notifications';
import { SCREENSHOT_DIR } from '../browser/screenshotStore';

export interface BillPayResult {
  success: boolean;
  approvalId?: string;
  preScreenshotPath?: string;
  postScreenshotPath?: string;
  error?: string;
  stage: 'lookup' | 'prepare' | 'approval' | 'submit' | 'confirm' | 'complete';
}

/**
 * Execute a full bill payment pipeline.
 *
 * @param billName - Human-friendly bill name (e.g., "Duke Energy")
 * @param serviceLink - Payment portal URL from Notion
 * @param notionBillId - Notion page ID for status updates
 * @param amount - Optional expected amount (from Notion). Actual amount is on the portal.
 */
export async function executeBillPayment(opts: {
  billName: string;
  serviceLink: string;
  notionBillId: string;
  amount?: number;
}): Promise<BillPayResult> {
  const { billName, serviceLink, notionBillId, amount } = opts;
  const cwd = process.cwd();

  console.log(`[BillPay] Starting payment workflow for ${billName} at ${serviceLink}`);

  // -----------------------------------------------------------------------
  // Stage 1: Prepare payment (form-filler sub-agent)
  //   Navigate to portal, authenticate, fill payment form.
  //   CRITICAL: Sub-agent must NOT click Submit/Pay. Take screenshot and stop.
  // -----------------------------------------------------------------------

  let preScreenshotPath: string | undefined;

  try {
    const agents = buildSubAgents();

    const preparePrompt = `Navigate to ${serviceLink} and prepare a payment for ${billName}.

STEPS:
1. Navigate to the URL: ${serviceLink}
2. If a login page appears, use the Bitwarden vault to retrieve credentials for this site and log in
3. Find the payment page or "Make a Payment" option
4. If there is a payment amount field, ${amount ? `enter $${amount.toFixed(2)}` : 'note the displayed amount'}
5. Take a screenshot of the filled payment form showing the amount to be paid

CRITICAL RULES:
- DO NOT click Submit, Pay, Pay Now, Confirm Payment, or any final submission button
- DO NOT proceed past the payment form — stop after filling and screenshotting
- Take a screenshot AFTER filling the form (this screenshot will be shown to the user for approval)
- If you encounter CAPTCHA, 2FA, or bot detection: STOP and return "BLOCKED: [reason]" with screenshot path
- Report the screenshot file path in your response`;

    let prepareResponse = '';
    const prepareConversation = query({
      prompt: preparePrompt,
      options: {
        cwd,
        systemPrompt: 'You are a form-filling agent preparing a bill payment. Follow instructions exactly.',
        allowedTools: ['mcp__bitwarden__*', 'mcp__playwright__*'],
        agents,
        permissionMode: 'bypassPermissions' as const,
        maxTurns: 15,
      },
    });

    for await (const event of prepareConversation) {
      if (event.type === 'result') {
        prepareResponse = (event as { type: 'result'; data: string }).data || '';
      }
    }

    // Check if sub-agent was blocked
    if (prepareResponse.includes('BLOCKED:')) {
      await sendTelegramAlert(
        `Bill payment blocked for ${billName}.\n\n${prepareResponse.match(/BLOCKED:\s*(.+?)(?:\n|$)/)?.[1] || 'Unknown reason'}\n\nPlease complete manually or retry later.`
      ).catch(() => {});

      return {
        success: false,
        error: `Browser automation blocked: ${prepareResponse.match(/BLOCKED:\s*(.+?)(?:\n|$)/)?.[1] || 'Unknown'}`,
        stage: 'prepare',
      };
    }

    // Extract screenshot path from response
    const screenshotPaths = prepareResponse.match(/[A-Z]:[/\\][^\s"']+\.png/gi) || [];
    preScreenshotPath = screenshotPaths[screenshotPaths.length - 1]; // Most recent screenshot

    if (!preScreenshotPath) {
      return {
        success: false,
        error: 'Form-filler did not produce a pre-submission screenshot. Cannot proceed without visual confirmation.',
        stage: 'prepare',
      };
    }

    console.log(`[BillPay] Pre-submission screenshot: ${preScreenshotPath}`);

  } catch (prepareErr) {
    const errMsg = (prepareErr as Error).message;
    console.error('[BillPay] Prepare stage failed:', errMsg);
    return {
      success: false,
      error: `Prepare stage failed: ${errMsg}`,
      stage: 'prepare',
    };
  }

  // -----------------------------------------------------------------------
  // Stage 2: Request human approval via approval gateway
  // -----------------------------------------------------------------------

  console.log(`[BillPay] Requesting approval for ${billName} payment...`);

  const approvalResult = await requestApproval({
    action: `Pay ${billName}${amount ? ` — $${amount.toFixed(2)}` : ''}\n\nReview the screenshot carefully. The amount shown on the portal is the actual charge.`,
    screenshotPath: preScreenshotPath,
    metadata: {
      domain: 'bill-pay',
      billName,
      notionBillId,
      amount,
      serviceLink,
    },
    timeoutMs: 10 * 60 * 1000, // 10 minutes
  });

  if (!approvalResult.approved) {
    console.log(`[BillPay] Payment ${approvalResult.status} for ${billName}`);
    return {
      success: false,
      approvalId: approvalResult.approvalId,
      preScreenshotPath,
      error: `Payment ${approvalResult.status} by user`,
      stage: 'approval',
    };
  }

  console.log(`[BillPay] Approved! Submitting payment for ${billName}...`);

  // -----------------------------------------------------------------------
  // Stage 3: Submit payment (browser-worker sub-agent)
  //   Click the submit/pay button and capture receipt screenshot.
  // -----------------------------------------------------------------------

  let postScreenshotPath: string | undefined;

  try {
    const agents = buildSubAgents();

    const submitPrompt = `The user has approved the payment. Now submit it.

STEPS:
1. Click the Submit, Pay, Pay Now, or Confirm Payment button on the current page
2. Wait for the confirmation page to load (wait 3-5 seconds after clicking)
3. Take a screenshot of the confirmation/receipt page
4. Report the confirmation screenshot file path in your response

CRITICAL RULES:
- Only click the submit button ONCE — do not retry if the page seems slow
- If the payment appears to fail (error message on page), take a screenshot and report the error
- If you encounter CAPTCHA or 2FA at this stage: STOP and return "BLOCKED: [reason]"
- Report the screenshot file path in your response`;

    let submitResponse = '';
    const submitConversation = query({
      prompt: submitPrompt,
      options: {
        cwd,
        systemPrompt: 'You are a browser automation agent submitting an approved payment. Follow instructions exactly.',
        allowedTools: ['mcp__playwright__*'],
        agents,
        permissionMode: 'bypassPermissions' as const,
        maxTurns: 10,
      },
    });

    for await (const event of submitConversation) {
      if (event.type === 'result') {
        submitResponse = (event as { type: 'result'; data: string }).data || '';
      }
    }

    // Extract post-submission screenshot
    const screenshotPaths = submitResponse.match(/[A-Z]:[/\\][^\s"']+\.png/gi) || [];
    postScreenshotPath = screenshotPaths[screenshotPaths.length - 1];

    if (submitResponse.includes('BLOCKED:')) {
      await sendTelegramAlert(
        `Payment submission blocked for ${billName} AFTER approval.\n\nThis needs manual completion — the payment form was filled and approved but submission hit a wall.`
      ).catch(() => {});

      return {
        success: false,
        approvalId: approvalResult.approvalId,
        preScreenshotPath,
        postScreenshotPath,
        error: `Submission blocked: ${submitResponse.match(/BLOCKED:\s*(.+?)(?:\n|$)/)?.[1] || 'Unknown'}`,
        stage: 'submit',
      };
    }

    console.log(`[BillPay] Payment submitted for ${billName}. Receipt: ${postScreenshotPath || 'no screenshot'}`);

  } catch (submitErr) {
    const errMsg = (submitErr as Error).message;
    console.error('[BillPay] Submit stage failed:', errMsg);

    await sendTelegramAlert(
      `Payment submission error for ${billName} AFTER approval.\n\nError: ${errMsg}\n\nPlease verify manually whether the payment went through.`
    ).catch(() => {});

    return {
      success: false,
      approvalId: approvalResult.approvalId,
      preScreenshotPath,
      error: `Submit stage failed: ${errMsg}`,
      stage: 'submit',
    };
  }

  // -----------------------------------------------------------------------
  // Stage 4: Log confirmation and send receipt
  // -----------------------------------------------------------------------

  // Send receipt screenshot to Telegram
  if (postScreenshotPath) {
    const { sendTelegramScreenshot } = await import('../telegram/notifications');
    await sendTelegramScreenshot(
      postScreenshotPath,
      `Payment confirmed for ${billName}. Receipt screenshot attached.`
    ).catch(() => {});
  }

  return {
    success: true,
    approvalId: approvalResult.approvalId,
    preScreenshotPath,
    postScreenshotPath,
    stage: 'complete',
  };
}
```

    case 'pay_bill': {
      const billName = input.bill_name as string;
      const requestedAmount = input.amount as number | undefined;
      const dataSourceId = LIFE_OS_DATABASES.subscriptions;
      if (!dataSourceId) {

      const page = await retrievePage(foundId);
      const props = (page as { properties: Record<string, unknown> }).properties;

      const titleProp = props[SUBSCRIPTION_PROPS.title] as { title?: Array<{ plain_text?: string }> } | undefined;
      const title = titleProp?.title?.[0]?.plain_text || billName;

      if (!serviceLink) {
        return `Found "${title}" but it doesn't have a payment link saved. You can say "update ${title} payment link to https://..." to add one, or use "navigate to payment for ${title}" to open the portal manually.`;
      }

      // Execute the bill payment workflow
      const { executeBillPayment } = await import('../workflows/billPayWorkflow');
      const result = await executeBillPayment({
        billName: title,
        serviceLink,
        notionBillId: foundId,
        amount: requestedAmount,
      });

      if (result.success) {
        // Mark the bill as paid in Notion
        try {
          const { buildBillPaidUpdate } = await import('./schemas');
          const paidProperties = buildBillPaidUpdate();
          await updatePage(foundId, paidProperties);
        } catch (notionErr) {
          console.error('[ToolExecutor] Failed to mark bill as paid in Notion:', notionErr);
          // Don't fail the whole operation -- the payment went through
        }

        return `Payment completed for "${title}". ${result.postScreenshotPath ? 'Receipt screenshot sent to Telegram.' : 'Confirmation captured.'} Bill marked as paid in Notion.`;
      } else {
        return `Payment workflow stopped at ${result.stage} stage for "${title}": ${result.error || 'Unknown error'}. ${result.stage === 'approval' ? 'You can retry when ready.' : 'Please check manually.'}`;
      }
    }

Key design decisions:
- The workflow uses `query()` from `@anthropic-ai/claude-code` directly (not through ccodeBrain.ts) because it needs specific sub-agent control (custom prompts, restricted tools, turn limits).
- Stage 1 (prepare) uses the form-filler sub-agent because it needs vault credentials for login.
- Stage 3 (submit) uses browser-worker because it only needs to click a button (no vault needed).
- The CRITICAL instruction "DO NOT click Submit" in the prepare prompt is the primary defense against Pitfall 3 (sub-agent eagerly submitting). maxTurns=15 is a secondary guard.
- Receipt screenshot is sent to Telegram as a confirmation (fire-and-forget).
- Notion bill status update is in a try/catch so payment success isn't masked by a Notion API failure.
- buildSubAgents() is called twice (once per sub-agent stage) to ensure fresh BW_SESSION.
