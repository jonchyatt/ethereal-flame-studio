```typescript
/**
 * Approval Gateway Types -- Jarvis v5.0
 *
 * Domain-agnostic type definitions for the approval system.
 * The gateway does not know about bills, payments, grants, or any specific domain.
 * Consumers (bill-pay, grant-submit, etc.) pass action descriptions and metadata.
 */

/** Status progression: pending -> approved | rejected | timeout | error */
export type ApprovalStatus = 'pending' | 'approved' | 'rejected' | 'timeout' | 'error';

/** Options passed to requestApproval() by any workflow */
export interface ApprovalRequestOptions {
  /** Human-readable description of the action requiring approval. E.g. "Pay Duke Energy $162.00" */
  action: string;
  /** Absolute path to a screenshot showing what is about to happen. Strongly recommended. */
  screenshotPath?: string;
  /** Arbitrary metadata for audit trail. Domain-specific data goes here. */
  metadata?: Record<string, unknown>;
  /** Timeout in ms before auto-reject. Default: 10 minutes. */
  timeoutMs?: number;
}

/** Result returned by requestApproval() after the deferred Promise resolves */
export interface ApprovalResult {
  approved: boolean;
  approvalId: string;
  status: ApprovalStatus;
  /** How long Jon took to respond, in ms. Null if timed out. */
  responseTimeMs: number | null;
}

/** Internal state for a pending approval (not exported to consumers) */
export interface PendingApproval {
  id: string;
  action: string;
  screenshotPath?: string;
  metadata?: Record<string, unknown>;
  resolve: (result: ApprovalResult) => void;
  timer: NodeJS.Timeout;
  messageId?: number;
  processed: boolean;
  requestedAt: number; // Date.now() timestamp
}

/** Shape of an audit trail entry for querying */
export interface ApprovalAuditEntry {
  id: string;
  action: string;
  status: ApprovalStatus;
  metadata?: string; // JSON string
  screenshotPath?: string;
  telegramMessageId?: number;
  requestedAt: string; // ISO timestamp
  respondedAt?: string; // ISO timestamp
  responseTimeMs?: number;
  createdAt: string;
}
```

```typescript
/**
 * Approval Audit Trail -- Jarvis v5.0
 *
 * Logs every approval lifecycle event to SQLite. This is the legal record
 * when Jarvis handles grant money, pays bills, or submits applications.
 *
 * Every request, response, timeout, and error is captured with timestamps.
 */

import { eq, desc } from 'drizzle-orm';
import { getDataDb } from '../data/db';
import { approvalAudit } from '../data/schema';
import type { ApprovalStatus } from './approvalTypes';

/**
 * Log a new approval request. Called when requestApproval() sends the Telegram message.
 */
export async function logApprovalEvent(entry: {
  id: string;
  action: string;
  status: ApprovalStatus;
  metadata?: Record<string, unknown>;
  screenshotPath?: string;
  telegramMessageId?: number;
}): Promise<void> {
  const db = getDataDb();
  await db.insert(approvalAudit).values({
    id: entry.id,
    action: entry.action,
    status: entry.status,
    metadata: entry.metadata ? JSON.stringify(entry.metadata) : null,
    screenshotPath: entry.screenshotPath || null,
    telegramMessageId: entry.telegramMessageId || null,
    requestedAt: new Date().toISOString(),
  });
}

/**
 * Update an existing approval record's status. Called on approve, reject, timeout, or error.
 */
export async function updateApprovalStatus(
  id: string,
  status: ApprovalStatus,
  extra?: { responseTimeMs?: number; errorMessage?: string }
): Promise<void> {
  const db = getDataDb();
  const now = new Date().toISOString();

  const updates: Record<string, unknown> = {
    status,
    updatedAt: now,
  };

  if (status === 'approved' || status === 'rejected' || status === 'timeout') {
    updates.respondedAt = now;
  }
  if (extra?.responseTimeMs !== undefined) {
    updates.responseTimeMs = extra.responseTimeMs;
  }
  if (extra?.errorMessage) {
    updates.errorMessage = extra.errorMessage;
  }

  await db.update(approvalAudit)
    .set(updates)
    .where(eq(approvalAudit.id, id));
}

/**
 * Get recent approval history, newest first. Useful for debugging and audit review.
 */
export async function getApprovalHistory(limit = 20): Promise<Array<typeof approvalAudit.$inferSelect>> {
  const db = getDataDb();
  return db.select()
    .from(approvalAudit)
    .orderBy(desc(approvalAudit.createdAt))
    .limit(limit);
}
```

Key design decisions:
- The approval_audit table uses TEXT primary key (UUID) not auto-increment integer, because the approval ID is generated by the gateway before the DB write (needed for Telegram callback data).
- metadata is a JSON string column so any domain can store arbitrary context (bill details, grant details, etc.) without schema changes.
- The audit module uses Drizzle ORM (same pattern as taskStore.ts), not raw SQL.
- logApprovalEvent and updateApprovalStatus are separate functions so the gateway can log at creation time and update on resolution without race conditions.

**Create `src/lib/jarvis/approval/approvalGateway.ts`:**

This is the core engine. It MUST be domain-agnostic. It receives action descriptions, screenshots, and metadata -- it does NOT know what a "bill" or "payment" is.

```typescript
/**
 * Approval Gateway -- Jarvis v5.0
 *
 * Domain-agnostic safety mechanism for sensitive actions.
 * Sends a Telegram approval request (photo + inline keyboard) and returns
 * a Promise that resolves when the user approves, rejects, or times out.
 *
 * Consumers: bill-pay workflow, grant submissions, credit applications, etc.
 * This module does NOT know about any specific domain.
 *
 * Safety invariants:
 * 1. Fail-safe default: timeout, send failure, or any ambiguity = REJECT
 * 2. Idempotent: double-tap on Approve does not double-process
 * 3. Audited: every event logged to SQLite via approvalAudit
 */

import { InlineKeyboard, InputFile } from 'grammy';
import { getTelegramBot } from '../telegram/bot';
import { logApprovalEvent, updateApprovalStatus } from './approvalAudit';
import type {
  ApprovalRequestOptions,
  ApprovalResult,
  PendingApproval,
} from './approvalTypes';
import * as fs from 'fs/promises';

/** In-memory map of pending approvals. Cleared on process restart (safe -- timeout = reject). */
const pendingApprovals = new Map<string, PendingApproval>();

/** Default timeout: 10 minutes */
const DEFAULT_TIMEOUT_MS = 10 * 60 * 1000;

/**
 * Request human approval for a sensitive action.
 *
 * Sends a Telegram message with an optional screenshot and inline Approve/Reject buttons.
 * Returns a Promise that resolves when the user responds or the timeout expires.
 *
 * FAIL-SAFE: If Telegram send fails, timeout fires, or anything goes wrong,
 * the result is { approved: false }. Never proceeds on ambiguous state.
 *
 * @example
 * const result = await requestApproval({
 *   action: 'Pay Duke Energy $162.00',
 *   screenshotPath: '/path/to/pre-submission.png',
 *   metadata: { billName: 'Duke Energy', amount: 162 },
 * });
 * if (result.approved) { // proceed with payment }
 */
export async function requestApproval(
  opts: ApprovalRequestOptions
): Promise<ApprovalResult> {
  const ownerId = process.env.TELEGRAM_OWNER_ID;
  if (!ownerId) {
    console.error('[ApprovalGateway] TELEGRAM_OWNER_ID not set -- auto-rejecting');
    return { approved: false, approvalId: 'no-owner', status: 'error', responseTimeMs: null };
  }

  const id = crypto.randomUUID();
  const timeoutMs = opts.timeoutMs ?? DEFAULT_TIMEOUT_MS;
  const requestedAt = Date.now();

  return new Promise<ApprovalResult>(async (resolve) => {
    // Helper to resolve exactly once
    const resolveOnce = (result: ApprovalResult) => {
      const pending = pendingApprovals.get(id);
      if (pending && !pending.processed) {
        pending.processed = true;
        clearTimeout(pending.timer);
        pendingApprovals.delete(id);
        resolve(result);
      }
    };

    // Set timeout -- auto-reject if no response (fail-safe)
    const timer = setTimeout(async () => {
      const responseTimeMs = Date.now() - requestedAt;
      resolveOnce({
        approved: false,
        approvalId: id,
        status: 'timeout',
        responseTimeMs,
      });

      // Update audit trail
      await updateApprovalStatus(id, 'timeout', { responseTimeMs }).catch((err) =>
        console.error('[ApprovalGateway] Failed to log timeout:', err)
      );

      // Remove buttons from Telegram message
      const pending = pendingApprovals.get(id) || { messageId: undefined };
      removeApprovalButtons(ownerId, pending.messageId, 'Timed out -- action cancelled.').catch(() => {});
    }, timeoutMs);

    // Register pending approval
    const approval: PendingApproval = {
      id,
      action: opts.action,
      screenshotPath: opts.screenshotPath,
      metadata: opts.metadata,
      resolve: resolveOnce as unknown as PendingApproval['resolve'],
      timer,
      processed: false,
      requestedAt,
    };
    pendingApprovals.set(id, approval);

    // Build inline keyboard
    const keyboard = new InlineKeyboard()
      .text('Approve', `approval:approve:${id}`)
      .text('Reject', `approval:reject:${id}`);

    const caption = `APPROVAL REQUIRED\n\n${opts.action}\n\nReview carefully, then tap Approve or Reject.`;

    try {
      const bot = getTelegramBot();
      let msg;

      if (opts.screenshotPath) {
        try {
          const buffer = await fs.readFile(opts.screenshotPath);
          msg = await bot.api.sendPhoto(
            ownerId,
            new InputFile(buffer, 'pre-action.png'),
            { caption: caption.slice(0, 1024), reply_markup: keyboard }
          );
        } catch (fileErr) {
          // Screenshot file unreadable -- fall back to text message
          console.warn('[ApprovalGateway] Screenshot unreadable, falling back to text:', fileErr);
          msg = await bot.api.sendMessage(ownerId, caption, { reply_markup: keyboard });
        }
      } else {
        msg = await bot.api.sendMessage(ownerId, caption, { reply_markup: keyboard });
      }

      approval.messageId = msg.message_id;

      // Log to audit trail
      await logApprovalEvent({
        id,
        action: opts.action,
        status: 'pending',
        metadata: opts.metadata,
        screenshotPath: opts.screenshotPath,
        telegramMessageId: msg.message_id,
      }).catch((err) => console.error('[ApprovalGateway] Failed to log approval event:', err));

    } catch (sendErr) {
      // If we can't send the approval request at all, auto-reject (fail-safe)
      console.error('[ApprovalGateway] Failed to send Telegram message -- auto-rejecting:', sendErr);
      resolveOnce({
        approved: false,
        approvalId: id,
        status: 'error',
        responseTimeMs: null,
      });

      await updateApprovalStatus(id, 'error', {
        errorMessage: `Telegram send failed: ${(sendErr as Error).message}`,
      }).catch(() => {});
    }
  });
}

/**
 * Handle an approval callback from Telegram.
 * Called by bot.ts callback_query:data handler when data matches 'approval:approve:*' or 'approval:reject:*'.
 *
 * Returns true if the callback was processed, false if unknown or already handled (idempotent).
 */
export function handleApprovalCallback(approvalId: string, approved: boolean): boolean {
  const pending = pendingApprovals.get(approvalId);
  if (!pending || pending.processed) return false;

  const responseTimeMs = Date.now() - pending.requestedAt;
  const status = approved ? 'approved' : 'rejected';

  // Resolve the deferred Promise (resumes the calling workflow)
  const result: ApprovalResult = {
    approved,
    approvalId,
    status,
    responseTimeMs,
  };

  pending.processed = true;
  clearTimeout(pending.timer);
  pendingApprovals.delete(approvalId);
  (pending.resolve as (r: ApprovalResult) => void)(result);

  // Update audit trail (fire-and-forget)
  updateApprovalStatus(approvalId, status, { responseTimeMs }).catch((err) =>
    console.error('[ApprovalGateway] Failed to log callback:', err)
  );

  return true;
}

/**
 * Get the count of currently pending approvals. Useful for health checks.
 */
export function getPendingApprovalCount(): number {
  return pendingApprovals.size;
}

/**
 * Remove inline keyboard buttons from a Telegram message (prevents re-clicks).
 * Optionally updates the caption/text.
 */
async function removeApprovalButtons(
  chatId: string,
  messageId?: number,
  newCaption?: string
): Promise<void> {
  if (!messageId) return;
  try {
    const bot = getTelegramBot();
    await bot.api.editMessageReplyMarkup(chatId, messageId, {
      reply_markup: { inline_keyboard: [] },
    });
    // Note: editMessageCaption could be used to update text, but it's complex
    // with photo vs text messages. Button removal is sufficient for safety.
  } catch {
    // Ignore -- message may have been deleted or already edited
  }
}
```
