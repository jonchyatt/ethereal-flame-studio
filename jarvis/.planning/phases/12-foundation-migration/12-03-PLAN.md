/**
 * Research Store — CRUD + Search for Research Entries
 *
 * Provides structured storage and retrieval of research findings.
 * Used by researcher sub-agent (save) and form-filler sub-agent (retrieve).
 *
 * Search: SQLite LIKE-based keyword search across topic, fieldName, fieldValue, notes, tags.
 * Semantic vector search will be layered on top via existing vectorSearch.ts infrastructure
 * when the research MCP tools are wired in Phase 16.
 *
 * Phase 12: Foundation schema + store
 */

import { eq, and, like, or, desc } from 'drizzle-orm';
import { getDataDb } from '../data/db';
import { researchEntries, type ResearchEntry, type NewResearchEntry } from '../data/schema';

// ---------------------------------------------------------------------------
// Save
// ---------------------------------------------------------------------------

export async function saveResearchEntry(
  entry: Omit<NewResearchEntry, 'id' | 'createdAt' | 'updatedAt'>
): Promise<ResearchEntry> {
  const db = getDataDb();
  const now = new Date().toISOString();
  const result = await db.insert(researchEntries).values({
    ...entry,
    createdAt: now,
    updatedAt: now,
  }).returning();
  return result[0];
}

// ---------------------------------------------------------------------------
// Search (keyword-based across multiple fields)
// ---------------------------------------------------------------------------

export async function searchResearch(query: string, domain?: string): Promise<ResearchEntry[]> {
  const db = getDataDb();
  const pattern = `%${query}%`;

  const keywordCondition = or(
    like(researchEntries.topic, pattern),
    like(researchEntries.fieldName, pattern),
    like(researchEntries.fieldValue, pattern),
    like(researchEntries.notes, pattern),
    like(researchEntries.tags, pattern),
  );

  const conditions = domain
    ? and(eq(researchEntries.domain, domain), keywordCondition)
    : keywordCondition;

  return db.select()
    .from(researchEntries)
    .where(conditions)
    .orderBy(desc(researchEntries.updatedAt))
    .limit(50);
}

// ---------------------------------------------------------------------------
// List topics (unique domain + topic combinations with entry counts)
// ---------------------------------------------------------------------------

export async function listResearchTopics(domain?: string): Promise<Array<{ domain: string; topic: string; count: number }>> {
  const db = getDataDb();

  // Fetch all entries for the domain (or all) and group in JS
  // SQLite via libsql doesn't have great GROUP BY support through Drizzle
  const condition = domain ? eq(researchEntries.domain, domain) : undefined;
  const entries = await db.select({
    domain: researchEntries.domain,
    topic: researchEntries.topic,
  }).from(researchEntries).where(condition);

  const counts = new Map<string, { domain: string; topic: string; count: number }>();
  for (const e of entries) {
    const key = `${e.domain}::${e.topic}`;
    const existing = counts.get(key);
    if (existing) {
      existing.count++;
    } else {
      counts.set(key, { domain: e.domain, topic: e.topic, count: 1 });
    }
  }

  return Array.from(counts.values()).sort((a, b) => a.domain.localeCompare(b.domain) || a.topic.localeCompare(b.topic));
}

// ---------------------------------------------------------------------------
// Get all entries for a specific topic
// ---------------------------------------------------------------------------

export async function getResearchByTopic(domain: string, topic: string): Promise<ResearchEntry[]> {
  const db = getDataDb();
  return db.select()
    .from(researchEntries)
    .where(and(eq(researchEntries.domain, domain), eq(researchEntries.topic, topic)))
    .orderBy(researchEntries.fieldName);
}

// ---------------------------------------------------------------------------
// Get single entry by ID
// ---------------------------------------------------------------------------

export async function getResearchEntry(id: number): Promise<ResearchEntry | undefined> {
  const db = getDataDb();
  const rows = await db.select().from(researchEntries).where(eq(researchEntries.id, id));
  return rows[0];
}

// ---------------------------------------------------------------------------
// Update
// ---------------------------------------------------------------------------

export async function updateResearchEntry(
  id: number,
  updates: Partial<Pick<ResearchEntry, 'fieldValue' | 'confidence' | 'notes' | 'tags' | 'expiresAt' | 'source'>>
): Promise<ResearchEntry | undefined> {
  const db = getDataDb();
  const result = await db.update(researchEntries)
    .set({ ...updates, updatedAt: new Date().toISOString() })
    .where(eq(researchEntries.id, id))
    .returning();
  return result[0];
}

// ---------------------------------------------------------------------------
// Delete
// ---------------------------------------------------------------------------

export async function deleteResearchEntry(id: number): Promise<boolean> {
  const db = getDataDb();
  const result = await db.delete(researchEntries).where(eq(researchEntries.id, id)).returning();
  return result.length > 0;
}

// ---------------------------------------------------------------------------
// Delete all entries for a topic (cleanup)
// ---------------------------------------------------------------------------

export async function deleteResearchTopic(domain: string, topic: string): Promise<number> {
  const db = getDataDb();
  const result = await db.delete(researchEntries)
    .where(and(eq(researchEntries.domain, domain), eq(researchEntries.topic, topic)))
    .returning();
  return result.length;
}
