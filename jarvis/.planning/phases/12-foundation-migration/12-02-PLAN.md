/**
 * Task Store — CRUD for Scheduled Tasks
 *
 * Provides add/edit/remove/list/toggle operations on the scheduled_tasks table.
 * Used by cronRunner.ts for loading tasks and by MCP tools for user management.
 */

import { eq } from 'drizzle-orm';
import { getDataDb } from '../data/db';
import { scheduledTasks, type ScheduledTask, type NewScheduledTask } from '../data/schema';

export async function listTasks(): Promise<ScheduledTask[]> {
  const db = getDataDb();
  return db.select().from(scheduledTasks);
}

export async function listEnabledTasks(): Promise<ScheduledTask[]> {
  const db = getDataDb();
  return db.select().from(scheduledTasks).where(eq(scheduledTasks.enabled, true));
}

export async function getTask(id: number): Promise<ScheduledTask | undefined> {
  const db = getDataDb();
  const rows = await db.select().from(scheduledTasks).where(eq(scheduledTasks.id, id));
  return rows[0];
}

export async function addTask(task: Omit<NewScheduledTask, 'id' | 'createdAt' | 'updatedAt'>): Promise<ScheduledTask> {
  const db = getDataDb();
  const now = new Date().toISOString();
  const result = await db.insert(scheduledTasks).values({
    ...task,
    createdAt: now,
    updatedAt: now,
  }).returning();
  return result[0];
}

export async function editTask(id: number, updates: Partial<Pick<ScheduledTask, 'name' | 'description' | 'cronExpression' | 'timezone' | 'handler' | 'config' | 'enabled'>>): Promise<ScheduledTask | undefined> {
  const db = getDataDb();
  const result = await db.update(scheduledTasks)
    .set({ ...updates, updatedAt: new Date().toISOString() })
    .where(eq(scheduledTasks.id, id))
    .returning();
  return result[0];
}

export async function removeTask(id: number): Promise<boolean> {
  const db = getDataDb();
  // Prevent deletion of system tasks
  const task = await getTask(id);
  if (!task) return false;
  if (task.isSystem) {
    throw new Error(`Cannot delete system task: ${task.name}`);
  }
  const result = await db.delete(scheduledTasks).where(eq(scheduledTasks.id, id)).returning();
  return result.length > 0;
}

export async function toggleTask(id: number): Promise<ScheduledTask | undefined> {
  const db = getDataDb();
  const task = await getTask(id);
  if (!task) return undefined;
  return editTask(id, { enabled: !task.enabled });
}

export async function updateTaskRun(id: number, result: string | null, error: string | null): Promise<void> {
  const db = getDataDb();
  await db.update(scheduledTasks)
    .set({
      lastRunAt: new Date().toISOString(),
      lastResult: result,
      lastError: error,
      updatedAt: new Date().toISOString(),
    })
    .where(eq(scheduledTasks.id, id));
}

export async function seedSystemTasks(): Promise<void> {
  const db = getDataDb();
  const existing = await db.select().from(scheduledTasks).where(eq(scheduledTasks.isSystem, true));
  if (existing.length > 0) return; // Already seeded

  const now = new Date().toISOString();
  await db.insert(scheduledTasks).values({
    name: 'Daily Reflection',
    description: 'Daily reflection loop + meta-evaluation at 5:00 AM ET',
    cronExpression: '0 5 * * *',
    timezone: 'America/New_York',
    handler: 'daily-reflection',
    enabled: true,
    isSystem: true,
    createdAt: now,
    updatedAt: now,
  });
}
```

/**
 * Cron Runner v2 — DB-Driven Flexible Scheduler
 *
 * Loads enabled tasks from scheduled_tasks table and schedules them via croner.
 * Hot-reloads every 60 seconds to pick up CRUD changes without restart.
 *
 * Replaces hardcoded node-cron with dynamic task management.
 * Existing hardcoded tasks are migrated as DB records with isSystem=true.
 *
 * Usage:
 *   npx tsx src/lib/jarvis/scheduler/cronRunner.ts
 */

import { Cron } from 'croner';
import { listEnabledTasks, updateTaskRun, seedSystemTasks } from './taskStore';
import { runReflection } from '../intelligence/reflectionLoop';
import { runMetaEvaluation } from '../intelligence/metaEvaluator';

// ---------------------------------------------------------------------------
// Handler Registry
// ---------------------------------------------------------------------------

const HANDLERS: Record<string, () => Promise<string>> = {
  'daily-reflection': async () => {
    let summary = '';

    try {
      const reflectionResult = await runReflection();
      summary += `Reflection: +${reflectionResult.rulesAdded} rules, ${reflectionResult.rulesSuperseded} superseded. `;
      if (reflectionResult.summary) {
        summary += reflectionResult.summary + ' ';
      }
    } catch (err) {
      summary += `Reflection failed: ${err instanceof Error ? err.message : String(err)}. `;
    }

    try {
      const metaResult = await runMetaEvaluation();
      if (metaResult?.ran) {
        summary += `Meta-evaluation: health ${metaResult.healthScore}/10.`;
      } else if (metaResult === null) {
        summary += 'Meta-evaluation: skipped (< 7 days since last).';
      }
    } catch (err) {
      summary += `Meta-evaluation failed: ${err instanceof Error ? err.message : String(err)}.`;
    }

    return summary;
  },
};

// ---------------------------------------------------------------------------
// Active cron instances (for hot-reload cleanup)
// ---------------------------------------------------------------------------

const activeCrons = new Map<number, Cron>();

// ---------------------------------------------------------------------------
// Core: Load tasks from DB and schedule
// ---------------------------------------------------------------------------

async function loadAndScheduleTasks(): Promise<void> {
  try {
    const tasks = await listEnabledTasks();

    // Stop all existing crons
    for (const cron of activeCrons.values()) {
      cron.stop();
    }
    activeCrons.clear();

    for (const task of tasks) {
      const handler = HANDLERS[task.handler];
      if (!handler) {
        console.warn(`[Cron] Unknown handler: ${task.handler} (task: ${task.name})`);
        continue;
      }

      const job = new Cron(task.cronExpression, { timezone: task.timezone }, async () => {
        console.log(`[Cron] Running: ${task.name}`);
        try {
          const result = await handler();
          console.log(`[Cron] ${task.name} complete: ${result}`);
          await updateTaskRun(task.id, result, null);
        } catch (err) {
          const msg = err instanceof Error ? err.message : String(err);
          console.error(`[Cron] ${task.name} failed:`, msg);
          await updateTaskRun(task.id, null, msg);
        }
      });

      activeCrons.set(task.id, job);
      console.log(`[Cron] Scheduled: ${task.name} (${task.cronExpression} ${task.timezone}) next: ${job.nextRun()?.toISOString() ?? 'N/A'}`);
    }

    console.log(`[Cron] ${activeCrons.size} task(s) active`);
  } catch (err) {
    console.error('[Cron] Failed to load tasks:', err);
  }
}

// ---------------------------------------------------------------------------
// Startup
// ---------------------------------------------------------------------------

async function main(): Promise<void> {
  console.log('[Cron] Jarvis cron runner v2 starting...');

  // Seed system tasks on first run
  await seedSystemTasks();

  // Initial load
  await loadAndScheduleTasks();

  // Hot-reload: poll for DB changes every 60 seconds
  setInterval(loadAndScheduleTasks, 60_000);

  console.log('[Cron] Hot-reload enabled (60s poll interval)');
}

main().catch((err) => {
  console.error('[Cron] Fatal error:', err);
  process.exit(1);
});

// Keep process alive
process.on('SIGINT', () => {
  console.log('[Cron] Shutting down...');
  for (const cron of activeCrons.values()) cron.stop();
  process.exit(0);
});
```
