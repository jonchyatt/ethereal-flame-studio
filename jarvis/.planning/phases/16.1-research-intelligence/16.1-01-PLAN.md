Add outcome tracking (which research led to approved applications), expiry monitoring (flag stale entries, notify for re-research), cross-domain inference (grant approval creates credit evidence), and confidence evolution (corroboration upgrades, contradiction downgrades).

Stale entries surface automatically. Grant successes strengthen credit applications. Confidence improves over time.

```typescript
export async function runCrossDomainInference(lookbackMs?: number): Promise<number>
```
- Default lookback: 24 hours
- Queries `research_outcomes WHERE outcome = 'approved' AND created_at > lookbackCutoff`
- For each outcome, iterates CROSS_DOMAIN_RULES
- Checks trigger.domain matches outcome.domain
- **Deduplication:** Before creating, search for existing entry with same domain+topic+fieldName where tags contain "auto-inferred". Skip if found.
- **Anti-circular:** Also skip if the source entry (from research_outcomes.researchEntryId) itself has tags containing "auto-inferred"
- Creates inferred entry via `saveResearchEntry()` with `source: 'auto-inferred from {outcome.domain} outcome'`
- Returns count of created entries

**Export `evolveConfidence()`** (INTEL-04):
```typescript
export async function evolveConfidence(): Promise<{ upgraded: number; downgraded: number; decayed: number; boosted: number }>
```
- Fetches all entries where confidence != 'expired'
- Groups by `${domain}::${topic}::${fieldName}`
- **Corroboration:** Groups where all entries have the same fieldValue but different sources (Set of sources >= 2) -- upgrade all to 'high'
- **Contradiction:** Groups where entries have different fieldValues -- downgrade all to 'low', add note "Conflicts with entry #{id}: {value}"
- **Age decay:** Entries older than 90 days with no linked outcomes, no updates in 90 days, and source is a URL (not 'grant-entity-profiles.md', not 'auto-inferred', not 'grant-application-workflow') -- downgrade one level (high->medium, medium->low). Skip entries with confidence='expired'.
- **Outcome boost:** Query research_outcomes for entries linked to approved outcomes. Upgrade those research entries to 'high' confidence.
- Returns counts of each operation

```typescript
'research-intelligence': async () => {
  // Step 1: Scan for expired entries
  const { stale, highValue } = await findStaleEntries();
  const expired = await markEntriesExpired(stale);

  // Step 2: Run confidence evolution
  const confidence = await evolveConfidence();

  // Step 3: Run cross-domain inference for recent outcomes
  const inferred = await runCrossDomainInference();

  // Step 4: Send Telegram summary if anything happened
  const parts: string[] = [];
  if (expired > 0) parts.push(`${expired} entries expired`);
  if (confidence.upgraded > 0) parts.push(`${confidence.upgraded} confidence upgrades`);
  if (confidence.downgraded > 0) parts.push(`${confidence.downgraded} confidence downgrades`);
  if (confidence.decayed > 0) parts.push(`${confidence.decayed} age decays`);
  if (confidence.boosted > 0) parts.push(`${confidence.boosted} outcome boosts`);
  if (inferred > 0) parts.push(`${inferred} cross-domain inferences`);

  const summary = parts.length > 0
    ? `Research Intelligence: ${parts.join(', ')}.`
    : 'Research Intelligence: no changes needed.';

  // Notify about high-value expired entries
  if (highValue.length > 0) {
    const hvList = highValue.map(e => `- ${e.domain}/${e.topic}: ${e.fieldName}`).join('\n');
    await sendTelegramAlert(
      `Research Intelligence Daily Report\n\n${summary}\n\nHigh-value entries need re-research:\n${hvList}`
    ).catch(() => {});
  } else if (parts.length > 0) {
    await sendTelegramAlert(`Research Intelligence Daily Report\n\n${summary}`).catch(() => {});
  }

  return summary;
},
```

```typescript
async function handleTrackOutcome(input: Record<string, unknown>): Promise<string> {
  const approvalId = input.approval_id as string;
  const domain = input.domain as string;
  const topic = input.topic as string;
  const outcome = input.outcome as string;
  const notes = input.notes as string | undefined;

  // Find all research entries for this domain+topic to link
  const entries = await getResearchByTopic(domain, topic);

  if (entries.length === 0) {
    return JSON.stringify({
      tracked: false,
      message: `No research entries found for ${domain}/${topic}. Track outcome manually or ensure research exists first.`,
    });
  }

  // Link each research entry to the outcome
  let linked = 0;
  for (const entry of entries) {
    await linkResearchOutcome({
      approvalId,
      researchEntryId: entry.id,
      outcome,
      domain,
      topic,
      notes,
    });
    linked++;
  }

  return JSON.stringify({
    tracked: true,
    approvalId,
    domain,
    topic,
    outcome,
    entriesLinked: linked,
  });
}

async function handleGetResearchInsights(input: Record<string, unknown>): Promise<string> {
  const insightType = input.insight_type as 'outcomes' | 'stale' | 'contradictions' | 'inferences' | 'all';
  const domain = input.domain as string | undefined;
  return getOutcomeInsights({ insightType, domain });
}
```
