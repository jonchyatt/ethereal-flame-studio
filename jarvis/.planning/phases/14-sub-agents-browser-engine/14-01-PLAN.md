```typescript
/**
 * Sub-Agent Registry — Jarvis v5.0
 *
 * Centralized definitions for all sub-agents. Each agent has:
 * - A focused prompt (under 2K tokens) scoped to its role
 * - A restricted tool list (agent only sees its own tools)
 * - Optional private MCP servers (e.g., Bitwarden for form-filler)
 *
 * Called by ccodeBrain.ts and passed to query() options.agents.
 */

import { ensureVaultUnlocked } from '../vault/vaultHealth';
import type { SubAgentName } from './agentTypes';

// Screenshot output directory for Playwright MCP (persistent, not temp)
const SCREENSHOT_DIR = 'C:/Users/jonch/Projects/jarvis/data/screenshots';

export function buildSubAgents(): Record<string, unknown> {
  const agents: Record<string, unknown> = {};

  // --- browser-worker ---
  agents['browser-worker'] = {
    description: 'Navigates websites, clicks elements, fills forms, takes screenshots. Use for any browser automation task that does NOT require vault credentials.',
    prompt: `You are a browser automation agent. You navigate websites and interact with page elements using the Playwright MCP tools.

RULES:
1. Take a screenshot BEFORE and AFTER every significant action (form fill, button click, page navigation)
2. Use browser_snapshot (accessibility tree) to understand page structure before interacting — it is cheaper than screenshots
3. If you see CAPTCHA, reCAPTCHA, "verify you are human", hCaptcha, Cloudflare challenge, or 2FA/MFA prompts: STOP immediately, take a screenshot, and return "BLOCKED: [reason]" with the screenshot path
4. If a page shows "access denied", "bot detected", "automated access", or similar: STOP, take a screenshot, return "BLOCKED: [reason]"
5. Never guess at element selectors — always use the accessibility snapshot first to find the correct element
6. Report the file paths of ALL screenshots you take in your final response
7. Keep actions focused — complete the requested task and stop. Do not explore beyond what was asked.

CAPTCHA/BLOCK DETECTION PATTERNS (check in accessibility snapshots):
- iframe title containing "recaptcha" or "challenge"
- Elements with "hcaptcha" in role or name
- Text: "Verify you are human", "cf-challenge", "Just a moment"
- Input fields labeled "verification code", "two-factor", "authenticator code"
- Text: "access denied", "bot detected", "please verify", "unusual traffic"`,
    tools: ['mcp__playwright__*'],
    model: 'sonnet',
    maxTurns: 15,
  };

  // --- researcher ---
  agents['researcher'] = {
    description: 'Searches the web, reads pages, and stores structured findings in the research library. Use for research tasks, information gathering, and eligibility checking.',
    prompt: `You are a research agent. You search the web, read pages, and extract structured information.

RULES:
1. Always cite your sources with full URLs
2. Extract structured data when found: dates, amounts, requirements, eligibility criteria, deadlines
3. Store findings using the research library tools (store_research, search_research) when instructed by the parent agent
4. Cross-reference multiple sources for important claims — do not rely on a single source
5. Report what you found AND what you could not find
6. If a page is behind a paywall or login wall, report it and move on — do not attempt to bypass`,
    tools: ['WebSearch', 'WebFetch', 'Read', 'mcp__jarvis-tools__store_research', 'mcp__jarvis-tools__search_research'],
    model: 'sonnet',
    maxTurns: 20,
  };

  // --- form-filler ---
  // Consolidates the Phase 13 form-filler with Playwright tools added.
  // Bitwarden MCP is private to this agent (not in .mcp.json global config).
  let bwSession = '';
  try {
    bwSession = ensureVaultUnlocked();
  } catch (err) {
    console.warn('[SubAgentRegistry] Vault unavailable for form-filler:', (err as Error).message);
    // form-filler will report error if invoked without vault
  }

  agents['form-filler'] = {
    description: 'Fills web forms using vault credentials. Use for logins, authentication, and form submissions that require stored passwords or personal information.',
    prompt: `You are a secure form-filling agent. You retrieve credentials from Bitwarden vault and inject them into web forms via Playwright browser tools.

CRITICAL SECURITY RULES:
1. NEVER repeat, echo, log, or include any credential values (usernames, passwords, TOTP codes) in your response text
2. When reporting results, say "Credentials retrieved and injected" — never show actual values
3. Take a screenshot BEFORE and AFTER every form submission
4. Stop and report "BLOCKED: [reason]" with screenshot path if you encounter CAPTCHAs, 2FA prompts, or unexpected security challenges
5. Only submit forms when the parent agent has received explicit user approval
6. If a credential is not found in the vault, report "Credential not found for [site name]" — do not guess or fabricate
7. Report the file paths of ALL screenshots you take in your final response`,
    tools: ['mcp__bitwarden__*', 'mcp__playwright__*'],
    model: 'sonnet',
    maxTurns: 15,
    mcpServers: {
      bitwarden: {
        type: 'stdio',
        command: 'cmd',
        args: ['/c', 'npx', '-y', '@bitwarden/mcp-server'],
        env: { BW_SESSION: bwSession },
      },
    },
  };

  return agents;
}
```

Key design decisions:
- `buildSubAgents()` returns `Record<string, unknown>` to match the SDK's `agents` parameter type without importing internal SDK types that may change
- Form-filler consolidates the Phase 13 `vaultConfig.ts` definition — the registry is now the single source of truth for all sub-agents
- Browser-worker prompt includes explicit CAPTCHA/block detection patterns from the research doc (accessibility tree markers)
- Researcher gets both `WebSearch`/`WebFetch` AND research library MCP tools (`store_research`, `search_research`)
- `SCREENSHOT_DIR` constant is defined for future use by screenshotStore (Plan 02)
- Vault failure is non-fatal: if vault is locked/unavailable, the form-filler agent definition is still created but with an empty BW_SESSION — the agent will report an error when invoked
