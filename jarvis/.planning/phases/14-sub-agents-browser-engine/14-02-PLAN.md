```typescript
/**
 * Screenshot Store — Jarvis v5.0
 *
 * Manages screenshots taken by browser sub-agents (Playwright MCP).
 * Screenshots are saved to a persistent directory for audit trails.
 * Provides utilities to list, retrieve, and clean up screenshot files.
 */

import * as fs from 'fs/promises';
import * as path from 'path';

/** Persistent screenshot directory — must match .mcp.json --output-dir and subAgentRegistry.ts */
export const SCREENSHOT_DIR = 'C:/Users/jonch/Projects/jarvis/data/screenshots';

export interface ScreenshotEntry {
  path: string;
  filename: string;
  createdAt: Date;
  sizeBytes: number;
}

/**
 * List all PNG screenshots in the screenshot directory, sorted newest first.
 */
export async function listScreenshots(): Promise<ScreenshotEntry[]> {
  try {
    const files = await fs.readdir(SCREENSHOT_DIR);
    const pngs = files.filter((f) => f.endsWith('.png'));

    const entries: ScreenshotEntry[] = [];
    for (const filename of pngs) {
      const fullPath = path.join(SCREENSHOT_DIR, filename);
      try {
        const stat = await fs.stat(fullPath);
        entries.push({
          path: fullPath,
          filename,
          createdAt: stat.mtime,
          sizeBytes: stat.size,
        });
      } catch {
        // File may have been deleted between readdir and stat
      }
    }

    // Sort newest first
    entries.sort((a, b) => b.createdAt.getTime() - a.createdAt.getTime());
    return entries;
  } catch {
    return [];
  }
}

/**
 * Get the most recent screenshot, optionally filtering by filename pattern.
 */
export async function getLatestScreenshot(pattern?: string): Promise<ScreenshotEntry | null> {
  const screenshots = await listScreenshots();
  if (pattern) {
    return screenshots.find((s) => s.filename.includes(pattern)) || null;
  }
  return screenshots[0] || null;
}

/**
 * Clean up screenshots older than the given age in hours.
 * Default: 24 hours. Returns number of files deleted.
 */
export async function cleanOldScreenshots(maxAgeHours = 24): Promise<number> {
  const cutoff = Date.now() - maxAgeHours * 60 * 60 * 1000;
  const screenshots = await listScreenshots();
  let deleted = 0;

  for (const entry of screenshots) {
    if (entry.createdAt.getTime() < cutoff) {
      try {
        await fs.unlink(entry.path);
        deleted++;
      } catch {
        // Ignore deletion failures
      }
    }
  }

  return deleted;
}
```

**Create `src/lib/jarvis/telegram/notifications.ts`:**

```typescript
/**
 * Telegram Notification Helpers — Jarvis v5.0
 *
 * Utilities for sending screenshots and alerts to Jon via Telegram.
 * Used by the parent agent when sub-agents report BLOCKED status
 * (CAPTCHA, 2FA, bot detection) during browser automation.
 *
 * All send functions are fire-and-forget: they log errors but never throw,
 * so the parent agent's response flow is not interrupted by Telegram failures.
 */

import { InputFile } from 'grammy';
import { getTelegramBot } from './bot';
import * as fs from 'fs/promises';

/**
 * Send a screenshot image to Jon via Telegram with a caption.
 *
 * @param screenshotPath - Absolute path to a PNG file on disk
 * @param caption - Description of what happened (max 1024 chars for Telegram)
 */
export async function sendTelegramScreenshot(
  screenshotPath: string,
  caption: string
): Promise<boolean> {
  const ownerId = process.env.TELEGRAM_OWNER_ID;
  if (!ownerId) {
    console.warn('[Telegram] No TELEGRAM_OWNER_ID set, cannot send screenshot notification');
    return false;
  }

  try {
    const bot = getTelegramBot();
    const fileBuffer = await fs.readFile(screenshotPath);
    await bot.api.sendPhoto(
      ownerId,
      new InputFile(fileBuffer, 'screenshot.png'),
      { caption: caption.slice(0, 1024) } // Telegram caption limit
    );
    console.log(`[Telegram] Screenshot sent: ${screenshotPath}`);
    return true;
  } catch (error) {
    console.error('[Telegram] Failed to send screenshot:', error);
    return false;
  }
}

/**
 * Send a text alert to Jon via Telegram.
 * Use for non-screenshot notifications (e.g., sub-agent completed, task failed).
 *
 * @param message - Alert text (max 4096 chars for Telegram)
 */
export async function sendTelegramAlert(message: string): Promise<boolean> {
  const ownerId = process.env.TELEGRAM_OWNER_ID;
  if (!ownerId) {
    console.warn('[Telegram] No TELEGRAM_OWNER_ID set, cannot send alert');
    return false;
  }

  try {
    const bot = getTelegramBot();
    await bot.api.sendMessage(ownerId, message.slice(0, 4096));
    return true;
  } catch (error) {
    console.error('[Telegram] Failed to send alert:', error);
    return false;
  }
}

/**
 * Parse a sub-agent BLOCKED response and send the appropriate Telegram notification.
 * Sub-agents return responses containing "BLOCKED: [reason]" and screenshot file paths.
 *
 * @param agentResponse - The full text response from a sub-agent
 * @param agentName - Which sub-agent reported the block (e.g., "browser-worker")
 * @returns true if a notification was sent, false if no BLOCKED pattern found or send failed
 */
export async function notifyIfBlocked(
  agentResponse: string,
  agentName: string
): Promise<boolean> {
  // Check for BLOCKED pattern in response
  const blockedMatch = agentResponse.match(/BLOCKED:\s*(.+?)(?:\n|$)/);
  if (!blockedMatch) return false;

  const reason = blockedMatch[1].trim();

  // Extract screenshot paths from response (Playwright MCP returns paths like C:/.../*.png)
  const screenshotPaths = agentResponse.match(/[A-Z]:[/\\][^\s"']+\.png/gi) || [];

  const caption = `Browser automation blocked (${agentName})\n\nReason: ${reason}\n\nAction needed: Please complete the challenge manually, then tell me to retry.`;

  // Send the most recent screenshot if available
  if (screenshotPaths.length > 0) {
    const lastScreenshot = screenshotPaths[screenshotPaths.length - 1];
    try {
      await fs.access(lastScreenshot);
      return await sendTelegramScreenshot(lastScreenshot, caption);
    } catch {
      // Screenshot file not accessible, fall back to text alert
    }
  }

  // No screenshot available — send text alert instead
  return await sendTelegramAlert(caption);
}
```

Key design decisions:
- All send functions return `boolean` and never throw — fire-and-forget pattern
- `notifyIfBlocked()` parses the sub-agent's BLOCKED response format to extract reason and screenshot paths
- Screenshot path regex matches Windows absolute paths (C:/... or C:\...)
- Caption is capped at 1024 chars (Telegram limit for photo captions)
- Text alerts capped at 4096 chars (Telegram message limit)
- `sendTelegramScreenshot` reads the file into a Buffer before sending (grammY InputFile pattern from research)
